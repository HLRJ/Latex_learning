配置texlive2023环境变量

\begin{lstlisting}
	vi ~/.bashrc
	export PATH=/usr/local/texlive/2023/bin/x86_64-linux:$PATH
	export MANPATH=/usr/local/texlive/2023/texmf-dist/doc/man:$MANPATH
	export INFOPATH=/usr/local/texlive/2023/texmf-dist/doc/info:$INFOPATH
\end{lstlisting}
